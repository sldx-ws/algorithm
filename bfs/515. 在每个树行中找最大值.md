[515. 在每个树行中找最大值](https://leetcode.cn/problems/find-largest-value-in-each-tree-row/)

给定一棵二叉树的根节点 `root` ，请找出该二叉树中每一层的最大值。

 

**示例1：**

![img](https://assets.leetcode.com/uploads/2020/08/21/largest_e1.jpg)

```
输入: root = [1,3,2,5,3,null,9]
输出: [1,3,9]
```

**示例2：**

```
输入: root = [1,2,3]
输出: [1,3]
```

 

**提示：**

- 二叉树的节点个数的范围是 `[0,10^4]`
- `-2^31 <= Node.val <= 2^31 - 1`



## 解法1：dfs

```c
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
#define max(a, b) ((a) > (b) ? (a) : (b))
 
int* ret;
int ret_len;

void dfs(struct TreeNode* root, int cur_depth)
{
    if (!root) return;

    ret[cur_depth] = max(ret[cur_depth], root->val);
    ret_len = max(ret_len, cur_depth + 1);

    dfs(root->left, cur_depth + 1);
    dfs(root->right, cur_depth + 1);
}
 
int* largestValues(struct TreeNode* root, int* returnSize) {
    ret = (int*)malloc(sizeof(int) * 10000);
    ret_len = 0;
    for (int i = 0; i < 10000; ++i) ret[i] = INT_MIN;
    
    dfs(root, 0);

    *returnSize = ret_len;
    return ret;
}
```

## 解法2：bfs

```c
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
#define max(a, b) ((a) > (b) ? (a) : (b))
 
int* largestValues(struct TreeNode* root, int* returnSize) {
    *returnSize = 0;
    if (!root) return NULL;

    int* ret = (int*)malloc(sizeof(int) * 10001);
    for (int i = 0; i < 10001; ++i) ret[i] = INT_MIN;

    struct TreeNode* queue[10001];
    int head = 0, tail = 0;
    queue[tail++] = root;

    while (head != tail)
    {
        int len = tail - head;

        for (int i = 0; i < len; ++i)
        {
            struct TreeNode* tmp = queue[head++];
            ret[*returnSize] = max(ret[*returnSize], tmp->val);

            if (tmp->left)
                queue[tail++] = tmp->left;
            
            if (tmp->right)
                queue[tail++] = tmp->right;
        }
        
        ++(*returnSize);
    }

    return ret;
}
```
