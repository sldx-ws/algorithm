[42. 接雨水](https://leetcode.cn/problems/trapping-rain-water/)

给定 `n` 个非负整数表示每个宽度为 `1` 的柱子的高度图，计算按此排列的柱子，下雨之后能接多少雨水。

 

**示例 1：**

![img](https://assets.leetcode.cn/aliyun-lc-upload/uploads/2018/10/22/rainwatertrap.png)

```
输入：height = [0,1,0,2,1,0,1,3,2,1,2,1]
输出：6
解释：上面是由数组 [0,1,0,2,1,0,1,3,2,1,2,1] 表示的高度图，在这种情况下，可以接 6 个单位的雨水（蓝色部分表示雨水）。 
```

**示例 2：**

```
输入：height = [4,2,0,3,2,5]
输出：9
```

 

**提示：**

- `n == height.length`
- `1 <= n <= 2 * 10^4`
- `0 <= height[i] <= 10^5`



## 解法1：dp

- 时间复杂度：O(n)

- 空间复杂度：O(n)

```c
int my_min(int a, int b)
{
    return a < b ? a : b;
}

int my_max(int a, int b)
{
    return a > b ? a : b;
}

int trap(int* height, int heightSize) {
    int maxL[heightSize];
    int maxR[heightSize];
    int ret = 0;

    maxL[0] = 0;
    for (int i = 1; i < heightSize; ++i)
    {
        maxL[i] = my_max(maxL[i - 1], height[i - 1]);
    }

    maxR[heightSize - 1] = 0;
    for (int i = heightSize - 2; i >= 0; --i)
    {
        maxR[i] = my_max(maxR[i + 1], height[i + 1]);
    }

    for (int i = 0; i < heightSize; ++i)
    {
        int tmp = my_min(maxL[i], maxR[i]) - height[i];
        if (tmp > 0)
        {
            ret += tmp;
        }
    }

    return ret;
}
```

## 解法2：双指针

- 时间复杂度：O(n)

- 空间复杂度：O(1)

```c
// 写法1
int my_max(int a, int b)
{
    return a > b ? a : b;
}

int trap(int* height, int heightSize) {
    int left = 0, right = heightSize - 1;
    int maxL = height[left], maxR = height[right];
    int ret = 0;

    while (left < right)
    {
        if (maxL < maxR)
        {
            ++left;
            maxL = my_max(maxL, height[left]);
            ret += maxL - height[left];
        }
        else 
        {
            --right;
            maxR = my_max(maxR, height[right]);
            ret += maxR - height[right];
        }
    }

    return ret;
}
```

// 写法2
int my_max(int a, int b)
{
    return a > b ? a : b;
}

int trap(int* height, int heightSize) {
    int maxL = height[0], maxR = height[heightSize - 1];
    int left = 0, right = heightSize - 1;
    int ret = 0;

    while (left <= right)
    {
        if (maxL < maxR)
        {
            if (maxL > height[left])
            {
                ret += maxL - height[left];
            }

            maxL = my_max(maxL, height[left]);
            ++left;
        }
        else 
        {
            if (maxR > height[right])
            {
                ret += maxR - height[right];
            }

            maxR = my_max(maxR, height[right]);
            --right;
        }
    }

    return ret;
}
