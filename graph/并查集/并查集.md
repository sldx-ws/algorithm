```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX 100
int father[MAX];

int find(int x)
{
    int root = x;
    while (father[root] >= 0) root = father[root];

    // 路径压缩
    while (father[x] >= 0)
    {
        int parent = father[x];
        father[x] = root;

        x = parent;
    }

    return root;
}

void _union(int x, int y)
{
    int rootx = find(x);
    int rooty = find(y);

    if (rootx == rooty) return;

    // 控制数量小的往大的合并
    if (father[rootx] < father[rooty])
    {
        father[rootx] += father[rooty];
        father[rooty] = rootx;
    }
    else
    {
        father[rooty] += father[rootx];
        father[rootx] = rooty;
    }	
}

int main()
{
    memset(father, -1, sizeof(father));
    
    _union(0, 1);
    _union(1, 2);
    _union(3, 4);

    printf("0和2连通？%d\n", find(0) == find(2)); // 1
    printf("0和3连通？%d\n", find(0) == find(3)); // 0

    _union(2, 3);
    printf("合并后0和3连通？%d\n", find(0) == find(3)); //1
   
    return 0;
}
```
