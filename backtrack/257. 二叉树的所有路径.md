[257. 二叉树的所有路径](https://leetcode.cn/problems/binary-tree-paths/)

给你一个二叉树的根节点 `root` ，按 **任意顺序** ，返回所有从根节点到叶子节点的路径。

**叶子节点** 是指没有子节点的节点。

**示例 1：**

![img](https://assets.leetcode.com/uploads/2021/03/12/paths-tree.jpg)

```
输入：root = [1,2,3,null,5]
输出：["1->2->5","1->3"]
```

**示例 2：**

```
输入：root = [1]
输出：["1"]
```

 

**提示：**

- 树中节点的数目在范围 `[1, 100]` 内
- `-100 <= Node.val <= 100`



## 解法（C）

```c
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
void dfs(struct TreeNode* root, char* path, int pathSize, char** ret, int* returnSize)
{
    if (root == NULL) return;

    pathSize += sprintf(path + pathSize, "%d", root->val);  // 注意sprintf返回值

    if (root->left == NULL && root->right == NULL)
    {
        // path[pathSize++] = '\0';
        pathSize++;  // sprintf自动在末尾加'\0'
        ret[*returnSize] = (char*)malloc(sizeof(char) * pathSize);
        strcpy(ret[*returnSize], path);
        ++(*returnSize);

        return;
    }
 
    pathSize += sprintf(path + pathSize, "->");
    dfs(root->left, path, pathSize, ret, returnSize);
    dfs(root->right, path, pathSize, ret, returnSize);
}

char** binaryTreePaths(struct TreeNode* root, int* returnSize) {
    char** ret = (char**)malloc(sizeof(char*) * 1001);
    *returnSize = 0;
    char* path = (char*)malloc(sizeof(char) * 1001);
    int pathSize = 0;

    dfs(root, path, pathSize, ret, returnSize);

    return ret;
}
```

### 补充

```c
int sprintf(char *str, const char *format, ...);
// sprintf会自动在末尾加'\0'
// 返回值：成功写入的字符数量（不包含末尾自动添加的'\0'）
```



## 解法（C++）

```cc
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<string> ret;

    void dfs(TreeNode* root, string path)
    {
        path += to_string(root->val);
        if (root->left == nullptr && root->right == nullptr)
        {
            ret.push_back(path);
            return;
        }

        path += "->";

        if (root->left)  dfs(root->left, path);  // 包含剪枝
        if (root->right) dfs(root->right, path); // 包含剪枝
        
    }

    vector<string> binaryTreePaths(TreeNode* root) {
        if (root == nullptr) return ret;
        dfs(root, "");

        return ret;
    }
};
```
