[131. 分割回文串](https://leetcode.cn/problems/palindrome-partitioning/)

给你一个字符串 `s`，请你将 `s` 分割成一些 子串，使每个子串都是 **回文串** 。返回 `s` 所有可能的分割方案。

 

**示例 1：**

```
输入：s = "aab"
输出：[["a","a","b"],["aa","b"]]
```

**示例 2：**

```
输入：s = "a"
输出：[["a"]]
```

 

**提示：**

- `1 <= s.length <= 16`
- `s` 仅由小写英文字母组成



## 解法

```c
/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
char** path;
int path_len;

char*** ret;
int* retColumnSizes;
int ret_len;

bool isPalindrome(char * s, int left, int right){
    while (left < right)
    {
        if (s[left] != s[right]) 
        {
            return false;
        }
        else 
        {
            ++left;
            --right;
        }
    }

    return true;
}

void backtracking(char*s, int startIndex)
{
    int len = strlen(s);

    if (startIndex == len)
    {
        ret[ret_len] = (char**)malloc(sizeof(char*) * path_len);
        for (int i = 0; i < path_len; ++i)
        {
            ret[ret_len][i] = path[i];
        }

        retColumnSizes[ret_len] = path_len;
        ++ret_len;
        
        return;
    }

    for (int i = startIndex; i < len; ++i)
    {
        if (!isPalindrome(s, startIndex, i)) continue;

        int subLen = i - startIndex + 1;
        path[path_len] = (char*)malloc(subLen + 1);
        strncpy(path[path_len], s + startIndex, subLen);
        path[path_len][subLen] = '\0';
        ++path_len;

        backtracking(s, i + 1);

        --path_len;
    }
}

char*** partition(char* s, int* returnSize, int** returnColumnSizes) {
    path = (char**)malloc(sizeof(char*) * 16);
    ret = (char***)malloc(sizeof(char**) * 200001);
    retColumnSizes = (int*)malloc(sizeof(int) * 200001);
    path_len = ret_len = 0;

    backtracking(s, 0);

    *returnSize = ret_len;
    *returnColumnSizes = retColumnSizes;
    
    return ret;
}
```
