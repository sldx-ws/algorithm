[40. 组合总和 II](https://leetcode.cn/problems/combination-sum-ii/)

给定一个候选人编号的集合 `candidates` 和一个目标数 `target` ，找出 `candidates` 中所有可以使数字和为 `target` 的组合。

`candidates` 中的每个数字在每个组合中只能使用 **一次** 。

**注意：**解集不能包含重复的组合。 

 

**示例 1:**

```
输入: candidates = [10,1,2,7,6,1,5], target = 8,
输出:
[
[1,1,6],
[1,2,5],
[1,7],
[2,6]
]
```

**示例 2:**

```
输入: candidates = [2,5,2,1,2], target = 5,
输出:
[
[1,2,2],
[5]
]
```

 

**提示:**

- `1 <= candidates.length <= 100`
- `1 <= candidates[i] <= 50`
- `1 <= target <= 30`



## 解法

```c
/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
int* path;
int path_len;

int** ret;
int* retColumnSizes;
int ret_len;

bool* used;

int cmp(const void* a, const void* b)
{
    return *(int*)a - *(int*)b;
}

void backtracking(int* candidates, int candidatesSize, int cur, int target, int sum)
{
    if (sum == target)
    {
        ret[ret_len] = (int*)malloc(sizeof(int) * path_len);
        for (int i = 0; i < path_len; ++i)
        {
            ret[ret_len][i] = path[i];
        }

        retColumnSizes[ret_len++] = path_len;
    }

    if (sum > target) return;

    for (int i = cur; i < candidatesSize; ++i)
    {
        if (i > cur && candidates[i] == candidates[i-1] && used[i - 1] == false) 
            continue;

        sum += candidates[i];
        path[path_len++] = candidates[i];
        used[i] = true;  // 数层去重

        backtracking(candidates, candidatesSize, i + 1, target, sum);

        sum -= candidates[i];
        --path_len;
        used[i] = false;
    }
}

int** combinationSum2(int* candidates, int candidatesSize, int target, int* returnSize, int** returnColumnSizes) {
    qsort(candidates, candidatesSize, sizeof(int), cmp);

    path = (int*)malloc(sizeof(int) * 100);
    ret = (int**)malloc(sizeof(int*) * 100);
    retColumnSizes = (int*)malloc(sizeof(int) * 100);
    path_len = ret_len = 0;

    used = (bool*)malloc(sizeof(bool) * 100);
    for (int i = 0; i < 100; ++i) used[i] = false;

    backtracking(candidates, candidatesSize, 0, target, 0);

    *returnSize = ret_len;
    *returnColumnSizes = retColumnSizes;

    return ret;
}
```
