[77. 组合](https://leetcode.cn/problems/combinations/)	

给定两个整数 `n` 和 `k`，返回范围 `[1, n]` 中所有可能的 `k` 个数的组合。

你可以按 **任何顺序** 返回答案。

 

**示例 1：**

```
输入：n = 4, k = 2
输出：
[
  [2,4],
  [3,4],
  [2,3],
  [1,2],
  [1,3],
  [1,4],
]
```

**示例 2：**

```
输入：n = 1, k = 1
输出：[[1]]
```

 

**提示：**

- `1 <= n <= 20`
- `1 <= k <= n`



## 解法

```c
/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
int** ans;
int ansSize;

int* path;
int pathSize;

void backtracking(int n, int k, int cur)
{
    if (pathSize == k)
    {
        ans[ansSize] = (int*)malloc(sizeof(int) * k);
        for (int i = 0; i < k; ++i) 
        {
            ans[ansSize][i] = path[i];
        }
        ++ansSize;

        return;
    }

    // for (int i = cur; i <= n; ++i)
    for (int i = cur; i <= n - (k - pathSize) + 1; ++i)  // 剪枝优化
    {
        path[pathSize++] = i;
        backtracking(n, k, i + 1);

        // 回溯
        --pathSize;
    }
}

int** combine(int n, int k, int* returnSize, int** returnColumnSizes) {
    path = (int*)malloc(sizeof(int) * k);
    ans = (int**)malloc(sizeof(int*) * 200001);
    pathSize = ansSize = 0;  // *全局变量赋值

    backtracking(n, k, 1);

    *returnSize = ansSize;
    *returnColumnSizes = (int*)malloc(sizeof(int) * (*returnSize));
    for (int i = 0; i < *returnSize; ++i) (*returnColumnSizes)[i] = k;

    return ans;
}
```
