有一个背包，最多能容纳的体积是V。

有n个物品，第i个物品的体积为`v[i]` ,价值为`w[i]`。

（1）背包最多能装多大价值的物品？

（2）若背包**恰好装满**，最多能装多大价值的物品？



**示例**

`输入`

```
3 5
2 10
4 5
1 4
```

`输出`

```
14
9
```



## 状态转移方程推导

### 背包不用恰好装满

- `dp[i][j]`
  - 不选`i`：`dp[i - 1][j]`
  - 选`1`个`i`：`dp[i - 1][j - v[i]] + w[i]`
- 即：`dp[i][j] = max(dp[i-1][j], dp[i-1][j-v[i]] + w[i])`

### 背包恰好装满

- 同上：`dp[i][j] = max(dp[i-1][j], dp[i-1][j-v[i]] + w[i])`
  - 用`-1`表示没有这个状态
  - 即：`dp[i-1][j-v[i]] != -1`

## 解法

```cc
#include <iostream>
#include <cstring>  // for memset
using namespace std;

const int N = 1010;
int dp[N][N];

int n; // 物品个数
int v[N]; // 物品体积
int w[N]; // 物品价值
int bagWeight; // 背包容量

int main()
{
    cin >> n >> bagWeight;
    for (int i = 1; i <= n; ++i) cin >> v[i] >> w[i];

    // (1)
    // dp[i][j]: 从前i个物品中挑选，总体积"不超过"j的所有选法中最大价值
    // 物品与背包遍历顺序 可交换
    memset(dp, 0, sizeof(dp));
    for (int i = 1; i <= n; ++i)  // 物品
    {
        for (int j = 0; j <= bagWeight; ++j)  // 背包
        {
            // 不选（01背包）i号物品
            dp[i][j] = dp[i-1][j];
            
            // 选1个（01背包）i号物品
            if (j - v[i] >= 0)
                dp[i][j] = max(dp[i][j], dp[i-1][j-v[i]] + w[i]);
        }
    }

    cout << dp[n][bagWeight] << endl;


    // (2)
    // dp[i][j]: 从前i个物品中挑选，总体积"正好等于"j的所有选法中最大价值
    // dp[i][j]==-1 表示没有这种情况
    // 物品与背包遍历顺序 可交换
    memset(dp, 0, sizeof(dp));
    for (int j = 1; j <= bagWeight; ++j) dp[0][j] = -1;

    for (int i = 1; i <= n; ++i)  // 物品
    {
        for (int j = 0; j <= bagWeight; ++j)  // 背包
        {
            // 不选（01背包）i号物品
            dp[i][j] = dp[i-1][j];
            
            // 选1个（01背包）i号物品
            if (j - v[i] >= 0 && dp[i-1][j-v[i]] != -1)
                dp[i][j] = max(dp[i][j], dp[i-1][j-v[i]] + w[i]);
        }
    }

    cout << (dp[n][bagWeight] == -1 ? 0 : dp[n][bagWeight]) << endl;

    return 0;
}
```

### 空间优化：滚动数组

**注意：遍历顺序必须从右向左**

```cc
#include <iostream>
#include <cstring>  // for memset
using namespace std;

const int N = 1010;
int dp[N];

int n; // 物品个数
int v[N]; // 物品体积
int w[N]; // 物品价值
int bagWeight; // 背包容量

int main()
{
    cin >> n >> bagWeight;
    for (int i = 1; i <= n; ++i) cin >> v[i] >> w[i];

    // (1)
    // 物品与背包遍历顺序 不可交换
    for (int i = 1; i <= n; ++i)
    {
    	for (int j = bagWeight; j - v[i] >= 0; --j) // 修改遍历顺序，必须从右向左
        {
            dp[j] = max(dp[j], dp[j-v[i]] + w[i]);
        }
    }
       
    cout << dp[bagWeight] << endl;


    // (2) - 写法1
    // 物品与背包遍历顺序 不可交换
    memset(dp, 0, sizeof(dp));
    for (int j = 1; j <= bagWeight; ++j) dp[j] = -1;

    for (int i = 1; i <= n; ++i)
    {
        for (int j = bagWeight; j - v[i] >= 0; --j) // 修改遍历顺序，必须从右向左
        {
            if (dp[j-v[i]] != -1)
                dp[j] = max(dp[j], dp[j-v[i]] + w[i]);
        }    
    }
    
    cout << (dp[bagWeight] == -1 ? 0 : dp[bagWeight]) << endl;

    return 0;
}
```
