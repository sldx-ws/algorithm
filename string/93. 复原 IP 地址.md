**有效 IP 地址** 正好由四个整数（每个整数位于 `0` 到 `255` 之间组成，且不能含有前导 `0`），整数之间用 `'.'` 分隔。

- 例如：`"0.1.2.201"` 和` "192.168.1.1"` 是 **有效** IP 地址，但是 `"0.011.255.245"`、`"192.168.1.312"` 和 `"192.168@1.1"` 是 **无效** IP 地址。

给定一个只包含数字的字符串 `s` ，用以表示一个 IP 地址，返回所有可能的**有效 IP 地址**，这些地址可以通过在 `s` 中插入 `'.'` 来形成。你 **不能** 重新排序或删除 `s` 中的任何数字。你可以按 **任何** 顺序返回答案。

 

**示例 1：**

```
输入：s = "25525511135"
输出：["255.255.11.135","255.255.111.35"]
```

**示例 2：**

```
输入：s = "0000"
输出：["0.0.0.0"]
```

**示例 3：**

```
输入：s = "101023"
输出：["1.0.10.23","1.0.102.3","10.1.0.23","10.10.2.3","101.0.2.3"]
```

 

**提示：**

- `1 <= s.length <= 20`
- `s` 仅由数字组成



## 解法

```c
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
char* path;
int path_len;

char** ret;
int ret_len;

bool isValid(char* s, int start, int end) {
    // 长度超过3 不合法
    if (end - start + 1 > 3) return false;
    // 前导零 不合法
    if (start != end && s[start] == '0') return false;
    
    // 转数字判断是否 <=255
    int num = 0;
    for (int i = start; i <= end; ++i) {
        num = num * 10 + (s[i] - '0');
    }
    return num <= 255;
}

void backtracking(char* s, int startIndex, int pointSum)
{
    int len = strlen(s);

    if (pointSum == 3)
    {
        if (startIndex < len && isValid(s, startIndex, strlen(s) - 1))
        {
            // for (int i = startIndex; i < len; i++) 
            // {
            //     path[path_len++] = s[i];
            // }
            strncpy(path + path_len, s + startIndex, len - startIndex);
            path_len += (len - startIndex);

            ret[ret_len] = (char*)malloc(sizeof(char) * (path_len + 1));
            // for (int i = 0; i < path_len; ++i)
            // {
            //     ret[ret_len][i] = path[i];
            // }
            // ret[ret_len][path_len] = '\0';
            strncpy(ret[ret_len], path, path_len);
            ret[ret_len][path_len] = '\0'; 

            ++ret_len;

            // 撤销最后一段
            path_len -= (len - startIndex);
        }

        return;
    }

    // for (int i = startIndex; i < len; ++i)
    for (int i = startIndex; i < len && i < startIndex + 3; ++i)  // 剪枝
    {
        if (isValid(s, startIndex, i))
        {
            int subLen = i - startIndex + 1;
            // for (int j = 0; j < subLen; ++j) 
            // {
            //     path[path_len++] = s[startIndex + j];
            // }
            strncpy(path + path_len, s + startIndex, subLen);
            
            path_len += subLen;
            path[path_len++] = '.';
            ++pointSum;

            backtracking(s, i + 1, pointSum);

            path_len -= (subLen + 1);
            --pointSum;
        }
    }
}

char** restoreIpAddresses(char* s, int* returnSize) {
    path = (char*)malloc(sizeof(char) * 16);
    ret = (char**)malloc(sizeof(char*) * 20001);
    path_len = ret_len = 0;

    backtracking(s, 0, 0);

    *returnSize = ret_len;

    return ret;
}
```
