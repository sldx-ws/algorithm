[22. 括号生成](https://leetcode.cn/problems/generate-parentheses/)

数字 `n` 代表生成括号的对数，请你设计一个函数，用于能够生成所有可能的并且 **有效的** 括号组合。

 

**示例 1：**

```
输入：n = 3
输出：["((()))","(()())","(())()","()(())","()()()"]
```

**示例 2：**

```
输入：n = 1
输出：["()"]
```

 

**提示：**

- `1 <= n <= 8`



## 解法

```c
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
char** ret;
int ret_len;

char* path;
int path_len;

void backtracking(int n, int left_bracket, int right_bracket)
{
    if (right_bracket == n)
    {
        ret[ret_len] = (char*)malloc(sizeof(char) * (path_len + 1));
        strcpy(ret[ret_len], path);
        ++ret_len;
        return;
    }
    
    if (left_bracket < n)
    {
        path[path_len++] = '(';
        backtracking(n, left_bracket + 1, right_bracket);
        --path_len;
    }

    if (right_bracket < left_bracket)
    {
        path[path_len++] = ')';
        backtracking(n, left_bracket, right_bracket + 1);
        --path_len;
    }
}
 
char** generateParenthesis(int n, int* returnSize) {
    ret = (char**)malloc(sizeof(char*) * 10001);
    path = (char*)malloc(sizeof(char) * (2 * n + 1));
    memset(path, 0, 2 * n + 1);
    ret_len = path_len = 0;

    backtracking(n, 0, 0);

    *returnSize = ret_len;
    return ret;
}
```
