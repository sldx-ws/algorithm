[51. N 皇后](https://leetcode.cn/problems/n-queens/)

按照国际象棋的规则，皇后可以攻击与之处在同一行或同一列或同一斜线上的棋子。

**n 皇后问题** 研究的是如何将 `n` 个皇后放置在 `n×n` 的棋盘上，并且使皇后彼此之间不能相互攻击。

给你一个整数 `n` ，返回所有不同的 **n 皇后问题** 的解决方案。

每一种解法包含一个不同的 **n 皇后问题** 的棋子放置方案，该方案中 `'Q'` 和 `'.'` 分别代表了皇后和空位。



**示例 1：**

![img](https://assets.leetcode.com/uploads/2020/11/13/queens.jpg)

```
输入：n = 4
输出：[[".Q..","...Q","Q...","..Q."],["..Q.","Q...","...Q",".Q.."]]
解释：如上图所示，4 皇后问题存在两个不同的解法。
```

**示例 2：**

```
输入：n = 1
输出：[["Q"]]
```

## 解法（C）

```c
/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
char** board;

char*** ret;
int ret_size;

bool isValid(int n, int x, int y)
{
    // 列
    for (int i = 0; i < x; ++i)
    {
        if (board[i][y] == 'Q') 
        {
            return false;
        }
    }

    // 45°
    for (int i = x - 1, j = y + 1; i >= 0 && j < n; --i, ++j)
    {
        if (board[i][j] == 'Q')
        {
            return false;
        }
    }

    // 135°
    for (int i = x - 1, j = y - 1; i >= 0 && j >= 0; --i, --j)
    {
        if (board[i][j] == 'Q')
        {
            return false;
        }
    }

    return true;
}

void backtracking(int n, int depth, int** returnColumnSizes)
{
    if (depth == n)
    {
        ret[ret_size] = (char**)malloc(sizeof(char*) * n);
        for (int i = 0; i < n; ++i)
        {
            ret[ret_size][i] = (char*)malloc(sizeof(char) * (n + 1));

            for (int j = 0; j < n; ++j)
            {
                ret[ret_size][i][j] = board[i][j];
            }

            ret[ret_size][i][n] = '\0';
        }

        (*returnColumnSizes)[ret_size++] = n;

        return;
    }

    //遍历横向棋盘
    for (int i = 0; i < n; ++i)
    {
        if (isValid(n, depth, i))
        {
            board[depth][i] = 'Q';

            backtracking(n, depth + 1, returnColumnSizes);
            
            board[depth][i] = '.';
        }
    }
}

char*** solveNQueens(int n, int* returnSize, int** returnColumnSizes) {
    board = (char**)malloc(sizeof(char*) * n);
    ret = (char***)malloc(sizeof(char**) * 400);
    *returnColumnSizes = (int*)malloc(sizeof(int) * 400);
    ret_size = 0;

    for (int i = 0; i < n; ++i)
    {
        board[i] = (char*)malloc(sizeof(char) * (n + 1));

        for (int j = 0; j < n; ++j)
        {
            board[i][j] = '.';
        }

        board[i][n] = '\0';
    }

    backtracking(n, 0, returnColumnSizes);

    *returnSize = ret_size;
    
    return ret;
}
```

## 解法（C++）

```cpp
class Solution {
public:

    int n;
    vector<vector<string>> ret;
    vector<string> board;

    bool isValid(int row, int col) 
    {
        for (int i = 0; i < row; ++i)  // 剪枝
            if (board[i][col] == 'Q') return false;

        for (int i = row-1, j = col+1; i >= 0 && j < n; --i, ++j)  // 45°
            if (board[i][j] == 'Q') return false;
            
        for (int i = row-1, j = col-1; i >= 0 && j >= 0; --i, --j) // 135°
            if (board[i][j] == 'Q') return false;
                             
        return true;
    }

    void dfs(int row) 
    {
        if (row == n) 
        {
            ret.push_back(board);
            return;
        }
        
        for (int col = 0; col < n; ++col) 
        {
            if (isValid(row, col)) 
            {
                board[row][col] = 'Q';
                dfs(row + 1);

                board[row][col] = '.'; // 恢复现场
            }
        }
    }

    vector<vector<string>> solveNQueens(int n_) {
        n = n_;  
        board = vector<string>(n_, string(n_, '.'));
        
        dfs(0);
        
        return ret;
    }
};
```

**优化-判断函数**

```CC
class Solution {
public:
    int n;
    vector<vector<string>> ret;
    vector<string> board;

    bool column[9] = { false };
    bool degree45[18] = { false };
    bool degree135[18] = { false };

    bool isValid(int row, int col) 
    {
        if (column[col] == false \
            && degree45[row + col] == false \
            && degree135[row - col + n] == false)
        return true;
                             
        return false;
    }

    void dfs(int row) 
    {
        if (row == n) 
        {
            ret.push_back(board);
            return;
        }
        
        for (int col = 0; col < n; ++col) 
        {
            if (isValid(row, col)) 
            {
                board[row][col] = 'Q';

                column[col] = true;             // 列占用
                degree45[row + col] = true; // 45°斜线已占用
                degree135[row - col + n] = true;    // 135°斜线已占用

                dfs(row + 1);

                // 恢复现场
                board[row][col] = '.'; 
                column[col] = false;
                degree45[row + col] = false;
                degree135[row - col + n] = false;
            }
        }
    }

    vector<vector<string>> solveNQueens(int n_) {
        n = n_;  
        board = vector<string>(n_, string(n_, '.'));
        
        dfs(0);
        
        return ret;
    }
};
```
